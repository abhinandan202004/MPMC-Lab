7.A C Program: Arithmetic Operations (ARM)


#include <lpc21xx.h>
main() {
    int a = 6, b = 2, sum, sub, mul, div;
    sum = a + b;
    sub = a - b;
    mul = a * b;
    div = a / b;
}


7.B C Program: Bitwise Operations (ARM)


#include <lpc21xx.h>
main() {
    int a = 10, b = 3, and, or, xor, not;
    and = a & b;
    or  = a | b;
    xor = a ^ b;
    not = ~a;
}
