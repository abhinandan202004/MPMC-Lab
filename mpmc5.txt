Date and Time Display (Assembly)
asm
Copy code
.MODEL SMALL
.DATA
MSG1 DB "TIME IS$"
MSG2 DB 10,13,"DATE IS$"
.CODE
MOV AX, @DATA
MOV DS, AX
LEA DX, MSG1
MOV AH, 09H
INT 21H
MOV AH, 2CH
INT 21H
MOV AL, CH
CALL DISPLAY
MOV AL, CL
CALL DISPLAY
MOV AL, DH
CALL DISPLAY
LEA DX, MSG2
MOV AH, 09H
INT 21H
MOV AH, 2AH
INT 21H
MOV BX, DX
MOV AL, DL
CALL DISPLAY
MOV AL, DH
CALL DISPLAY
MOV AX, CX
CALL DISPLAY

DISPLAY PROC
PUSH AX
MOV DL, ':'
MOV AH, 02H
INT 21H
POP AX
AAM
MOV DX, AX
ADD DX, 3030H
XCHG DH, DL
MOV AH, 02H
INT 21H
MOV DL, DH
MOV AH, 02H
INT 21H
RET
DISPLAY ENDP
END
